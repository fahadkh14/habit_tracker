version: "3.9"

services:

  mysql:

    image: mysql:8.4

    container_name: habit-mysql

    restart: always

    env_file:
      - .env

    ports:
      - "3307:3306"

    volumes:
      - mysql-data:/var/lib/mysql
      - ./database/init.sql:/docker-entrypoint-initdb.d/init.sql

    networks:
      - habit-network

    healthcheck:
      test: ["CMD-SHELL", "mysqladmin ping -h localhost -uroot -p$${MYSQL_ROOT_PASSWORD} --silent"]
      interval: 10s
      timeout: 5s
      retries: 5

  backend:

    build: ./backend

    container_name: habit-backend

    restart: always

    env_file:
      - .env

    depends_on:
      - mysql:

    ports:
      - "5000:5000"

    networks:
      - habit-network

  frontend:

    build: ./frontend

    container_name: habit-frontend

    restart: always

    depends_on:
      - backend

    ports:
      - "3000:80"

    networks:
      - habit-network

networks:
  habit-network:

volumes:
  mysql-data:
